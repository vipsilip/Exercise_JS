export function insert(arr, newItem, k) {
  if (!Array.isArray(arr) || arr.length === 0) return [];
  if (k <= 0) {
    arr.unshift(newItem);
    return arr;
  }
  if (k > arr.length) {
    arr.push(newItem);
    return arr;
  }
  arr.splice(k, 0, newItem);
  return arr;
}

export function remove(arr, k, n) {
  if (!Array.isArray(arr) || arr.length === 0) return [];
  if (k < 0 || k > arr.length - 1) return [...arr];
  if (n === undefined) {
    arr.splice(0, arr.length);
    return [...arr];
  }
  arr.splice(k, n);
  return [...arr];
}
